    function y = dquanti(x, FS, Nbits, gama)
    if gama == 0
        g_x = x; % Si gama=0, no hay distorsion (g(x) = x)
    else
        g_x = sign(x) .* (FS / log(1 + gama)) .* log(1 + gama .* abs(x) / FS);
        g_x(x == 0) = 0;
    end

    FS    = abs(FS);
    FSbin = 2^(Nbits-1);
    LSB   = FS/FSbin;  

        y = round(g_x/LSB); 
        y = min(y, FSbin-1); 
        y = max(y, -FSbin); 
        y = y * LSB;
